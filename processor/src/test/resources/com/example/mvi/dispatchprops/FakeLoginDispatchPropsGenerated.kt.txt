package com.example.mvi.dispatchprops

import com.example.mvi.entity.FormFieldValues
import java.util.LinkedList
import kotlin.String
import kotlin.collections.List

abstract class FakeLoginDispatchPropsGenerated : LoginDispatchProps {
    private val _methodCalls: LinkedList<Method> = LinkedList<Method>()

    val methodCalls: List<Method>
        get() = _methodCalls

    override fun callToSupport() {
        _methodCalls.add(Method.CallToSupport)
    }

    override fun startHelp(message: String) {
        _methodCalls.add(Method.StartHelp(message))
    }

    override fun startLogin(credentials: FormFieldValues) {
        _methodCalls.add(Method.StartLogin(credentials))
    }

    sealed class Method {
        object CallToSupport : Method()

        data class StartHelp(val message: String) : Method()

        data class StartLogin(val credentials: FormFieldValues) : Method()
    }
}
